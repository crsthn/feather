## Anatomy

```tsx
import {
  Menu,
  MenuTrigger,
  MenuPopup,
  MenuItem,
  MenuSeparator,
  MenuCheckboxItem,
  MenuRadioGroup,
  MenuRadioItem,
  MenuGroup,
  MenuGroupLabel,
  MenuSubmenuTrigger,
} from "@/components/ui/dropdown"
```

```tsx
<Menu>
  <MenuTrigger />
  <MenuPopup>
    <MenuGroup>
      <MenuGroupLabel />
      <MenuItem />
    </MenuGroup>
    <MenuSeparator />
    <MenuCheckboxItem />
    <MenuRadioGroup>
      <MenuRadioItem />
    </MenuRadioGroup>
    <Menu>
      <MenuSubmenuTrigger />
      <MenuPopup>
        <MenuItem />
      </MenuPopup>
    </Menu>
  </MenuPopup>
</Menu>
```
